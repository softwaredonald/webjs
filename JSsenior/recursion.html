命名函数表达式

                var f = function(){
                    console.log('sss');
                }
        
                f(); //调用函数表达式执行；




                //函数命名表达式。fun相当于在函数内部添加一个变量fun，fun指向函数自身
                // fun=== arguments.callee
                // fun的作用域在函数内部，不会影响函数外面的声明的参数或者函数。
                var m = function fun(){
                    console.log("fun");
                    fun();
                }
                m();

求num的阶乘，用递归实现 

            function f(num) {
                if (num <= 1) {
                    return 1;
                }
                return num*f(num-1)
    
            }

斐波那契数列

f(n)=f(n-1)+f(n-2) 

            function f(n){
                if(n==1){return 1;};
                if(n==0){return 0;};
                return f(n-1)+f(n-2);
            }
